37  import static liquibase.ext.Constants.EXTENSION_PRIORITY;

39  /**
40   * Overridden to remove schema dependent info
41   *
42   * @author Leo Przybylski (leo [at] rsmart.com)
43   */
44  public class CreateIndexChange extends liquibase.change.core.CreateIndexChange {
45  
46      public CreateIndexChange() {
47          setPriority(EXTENSION_PRIORITY);
48      }
49  
50      public String getSchemaName() {
51          return null;
52      }
53  
54      public SqlStatement[] generateStatements(Database database) {
55          List<String> columns = new ArrayList<String>();
56          for (ColumnConfig column : getColumns()) {
57              columns.add(column.getName());
58          }
59  
60  	    return new SqlStatement[]{
61  			    new CreateIndexStatement(
62  					    getIndexName(),
63  					    getSchemaName(),
64  					    getTableName(),
65  					    this.isUnique(),
66  					    getAssociatedWith(),
67  					    columns.toArray(new String[getColumns().size()]))
68  					    .setTablespace(getTablespace())
69  	    };
70      }
71  }
